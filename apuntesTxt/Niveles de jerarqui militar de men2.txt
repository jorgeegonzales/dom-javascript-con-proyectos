// Niveles de jerarqui militar de menor a mayor
//Cabo,Sargento,Teniente,Capitan,Mayor,Coronel,general

// Initialize constructor functions
function General(nameGeneral) {
  this.nameGeneral = nameGeneral;
}

function Coronel(nameCoronel,apeCoronel) {
  General.call(this, nameCoronel);
  this.apeCoronel = apeCoronel;
}

function Mayor(nameMayor,apeMayor) {
  General.call(this, nameMayor);
  this.apeMayor = apeMayor;
}

function Capitan(nameCapitan,apeCapitan) {
  General.call(this, nameCapitan);
  this.apeCapitan = apeCapitan;
}

function Teniente(nameTeniente,apeTeniente) {
  General.call(this, nameTeniente);
  this.apeTeniente = apeTeniente;
}

function Sargento(nameSargento,apeSargento) {
  General.call(this, nameSargento);
  this.apeSargento = apeSargento;
}

function Cabo(nameCabo,apeCabo) {
  General.call(this, nameCabo);
  this.apeCabo = apeCabo;
}

// Niveles de jerarqui militar de menor a mayor
//Cabo,Sargento,Teniente,Capitan,Mayor,Coronel,general

// Link prototypes and add prototype methods
Object.setPrototypeOf(Coronel.prototype, General.prototype);
Object.setPrototypeOf(Mayor.prototype, Coronel.prototype);
Object.setPrototypeOf(Capitan.prototype, Mayor.prototype);
Object.setPrototypeOf(Teniente.prototype, Capitan.prototype);
Object.setPrototypeOf(Sargento.prototype, Teniente.prototype);
Object.setPrototypeOf(Cabo.prototype, Sargento.prototype);

General.printGeneral = function () {
  return `Nombre: ${this.nameGeneral}.`;
}
